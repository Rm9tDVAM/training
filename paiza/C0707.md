import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        int n=sc.nextInt();sc.nextLine();
        String g=sc.nextLine();
        String temp;
        for(int i=0,check=0;i<n;i++){
            temp=sc.nextLine();
            if(temp.contains(g)){
                System.out.println(temp);
                check++;
            }
            if(i+1==n){
                if(check==0){
                    System.out.print("None");
                }
            }
        }
        sc.close();
    }
}
