import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        int n=sc.nextInt();sc.nextLine();
        String str="";
        String[] sArr=new String[n];
        for(int i=0;i<n;i++){
            sArr[i]=sc.nextLine();
        }
        for(int i=0;i<n;i++){
            String[] tArr=sArr[i].split("");
            Map<String,Integer> map=new HashMap<>();
            for(int j=0;j<tArr.length;j++){
                if(!(map.containsKey(tArr[j]))){
                    map.put(tArr[j],1);
                }else{
                    map.put(tArr[j],map.get(tArr[j])+1);
                }
            }
            if(map.size()==1){
                str="Four Card";
            }else if(map.size()==4){
                str="No Pair";
            }else if(map.size()==3){
                str="One Pair";
            }else if(map.size()==2){
                for(String key:map.keySet()){
                    if(map.get(key)==1){
                        str="Three Card";
                        break;
                    }else{
                        str="Two Pair";
                    }
                }
            }
            System.out.println(str);
        }
        sc.close();
    }
}
