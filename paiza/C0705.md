import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        int Q=sc.nextInt();
        int[] iArr=new int[Q];
        String str;
        for(int i=0;i<Q;i++){
            iArr[i]=sc.nextInt();
        }
        for(int i=0;i<iArr.length;i++){
            for(int j=0,sum=0;j<iArr[i]/2+1;j++){
                if(!(j==0)&&iArr[i]%j==0){
                    sum+=j;
                }
                if(j+1==iArr[i]/2+1){
                    if(sum==iArr[i]){
                        str="perfect";
                    }else if(sum==iArr[i]-1){
                        str="nearly";
                    }else{
                        str="neither";
                    }
                    System.out.println(str);
                }
            }
        }
        sc.close();
    }
}
