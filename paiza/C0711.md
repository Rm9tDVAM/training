                iArr[i][j]=sc.nextInt();
            }
        }
        System.out.print(iArr[0][0]+" ");
        System.out.print(iArr[iArr.length-1][1]+" ");
        for(int i=0,max=0,min=1000000;i<iArr.length;i++){
            max=Math.max(max,iArr[i][2]);
            min=Math.min(min,iArr[i][3]);
            if(i+1==iArr.length){
                System.out.print(max+" ");
                System.out.print(min);
            }
        }
        sc.close();
    }
}

import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        String sLine=sc.nextLine();
        String[] sArr=sLine.split(" ");
        List<String> list=new ArrayList<>();
        int count=0;
        String t=sArr[1];
        String s=sArr[2];
        while(!t.equals(s)){
            if(count==0){
                for(int i=0;i<s.length();i++){
                    list.add(s.substring(i,i+1));
                }
            }
            list.add(list.get(0));
            list.remove(0);
            s="";
            for(String l:list){
                s+=l;
            }
            count++;
        }
        System.out.print(count);
        sc.close();
    }
}

import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        final int N=sc.nextInt();sc.nextLine();
        int sum=0;
        String[] sArr=new String[N];
        for(int i=0;i<N;i++){
            sArr[i]=sc.nextLine();
        }
        for(int i=0;i<sArr.length;i++){
            String[] tArr=sArr[i].split(" ");
            String q=tArr[0];
            String a=tArr[1];
            if(!(q.length()==a.length())){
                sum+=0;
            }else if(q.equals(a)){
                sum+=2;
            }else{
                String[] qArr=q.split("");
                String[] aArr=a.split("");
                int err=0;
                for(int j=0;j<qArr.length;j++){
                    if(!qArr[j].equals(aArr[j])){
                        err++;
                    }
                }
                if(err==1){
                sum+=1;
                }
            }
        }
        System.out.print(sum);
        sc.close();
    }
}
