    ~~~java
    import java.util.*;
    import java.util.stream.*;
    public class Main{
        public static void main(String[] args){
            Scanner sc=new Scanner(System.in);
            int k=sc.nextInt();
            int n=sc.nextInt();sc.nextLine();
            String[] sArr=new String[k];
            for(int i=0;i<k;i++){
            sArr[i]=sc.nextLine();
            }
            for(int i=0;i<k;i++){
                List<Integer> iList=Arrays.stream(sArr[i].split(" "))
                    .map(Integer::parseInt)
                    .collect(Collectors.toList());
                int d=iList.get(0);
                int a=iList.get(1);
                int sum=0;
                if(d>=10){
                    sum=0;
                }else if(d>0){
                    sum=(int)Math.floor(((100/n)*a)*0.8);
                }else{
                    sum=(100/n)*a;
                }
                String ans="";
                if(sum>=80){
                    ans="A";
                }else if(sum>=70){
                    ans="B";
                }else if(sum>=60){
                    ans="C";
                }else if(sum<=59){
                    ans="D";
                }
                System.out.println(ans);
            }
            sc.close();
        }
    }
    ~~~

1. Java
    ~~~java
    import java.util.*;
    public class Main{
        public static void main(String[] args){
            Scanner sc=new Scanner(System.in);
            // 1<=N<=1000, 各レシートの数
            int N=sc.nextInt();sc.nextLine();
            String[] sArr=new String[N];
            for(int i=0;i<sArr.length;i++){
                sArr[i]=sc.nextLine();
            }
            // 1<=d_i<=31, 日付
            // 1<=p_i<=10000, 購入金額
            int point=0;
            for(int i=0;i<sArr.length;i++){
                String[] tArr=sArr[i].split(" ");
                String d_i=tArr[0];
                int p_i=Integer.parseInt(tArr[1]);
                if(d_i.contains("5")){
                    point+=(int)Math.floor(p_i*0.05);
                }else if(d_i.contains("3")){
                    point+=(int)Math.floor(p_i*0.03);
                }else{
                    point+=(int)Math.floor(p_i*0.01);
                }
            }
            System.out.print(point);
            sc.close();
        }
    }
    ~~~
