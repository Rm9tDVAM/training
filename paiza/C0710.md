import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        int a=sc.nextInt();
        int b=sc.nextInt();
        int len=0;
        if(String.valueOf(a).length()>String.valueOf(b).length()){
            len=String.valueOf(a).length();
        }else{
            len=String.valueOf(b).length();
        }
        int n=1;
        for(int i=1;i<len;i++){
            n*=10;
        }
        int[] aArr=new int[len];
        for(int i=0,temp=n;i<len;i++,temp/=10){
            aArr[i]=a/temp;
            a%=temp;
        }
        int[] bArr=new int[len];
        for(int i=0,temp=n;i<len;i++,temp/=10){
            bArr[i]=b/temp;
            b%=temp;
        }
        for(int i=0;i<len;i++){
            System.out.print((aArr[i]+bArr[i])%10);
        }
        sc.close();
    }
}

import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        double m=sc.nextDouble();
        double n=sc.nextDouble();

        double[] c=new double[(int)m];
        for(int i=0;i<c.length;i++){
            c[i]=sc.nextDouble()/100;
        }

        double[][] a=new double[(int)n][(int)m];
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                a[i][j]=sc.nextDouble();
            }
        }

        for(int i=0;i<n;i++){
            for(int j=0,temp=0;j<m;j++){
                temp+=(int)Math.floor(c[j]*a[i][j]);
                if(j+1==m){
                    System.out.println(temp);
                }
            }
        }
        sc.close();
    }
}
