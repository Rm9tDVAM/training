import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        int h=sc.nextInt();
        int w=sc.nextInt();
        int x=sc.nextInt();sc.nextLine();
        int range=0;
        String str="";
        for(int i=0;i<h;i++){
            str+=sc.nextLine();
        }
        if(str.length()%x>0){
            range=(str.length()/x)+1;
        }else{
            range=(str.length()/x);
        }
        for(int i=0,index1=0,index2=x;i<range;i++,index1+=x,index2+=x){
            if(i+1==range){
                System.out.println(str.substring(index1));
            }else{
                System.out.println(str.substring(index1,index2));
            }
        }
        sc.close();
    }
}

import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        int n=sc.nextInt();
        int c1=sc.nextInt();
        int c2=sc.nextInt();
        int stock=0;
        int sum=0;
        for(int i=0,temp=0;i<n;i++){
            temp=sc.nextInt();
            if(temp<=c1){
                sum-=temp;
                stock++;
            }else if(temp>=c2){
                sum+=temp*stock;
                stock=0;
            }
            if(i+1==n&&stock>0){
                sum+=temp*stock;
            }
        }
        System.out.print(sum);
        sc.close();
    }
}

import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        int n=sc.nextInt();
        int[] iArr=new int[n];
        int sum=0;
        for(int i=0;i<n;i++){
            iArr[i]=sc.nextInt();
        }
        for(int i=0,pos=1;i<iArr.length;i++){
            if(pos<iArr[i]){
                sum+=iArr[i]-pos;
            }else{
                sum+=pos-iArr[i];
            }
            pos=iArr[i];
        }
        System.out.print(sum);
        sc.close();
    }
}

import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        int k=sc.nextInt();
        int n=100/sc.nextInt();
        String str="";
        for(int i=0;i<k;i++){
            int score=0;
            int d=sc.nextInt();
            int a=sc.nextInt();
            if(d>=10){
                score=0;
            }else if(d>0){
                score=(int)Math.floor(a*n*0.8);
            }else{
                score=a*n;
            }

            if(score>=80){
                str="A";
            }else if(score>=70){
                str="B";
            }else if(score>=60){
                str="C";
            }else{
                str="D";
            }
            System.out.println(str);
        }
        sc.close();
    }
}

public class Main{
    public static void main(String[] args){

        int n=sc.nextInt();sc.nextLine();

        for(int i=0;i<n;i++){
            sArr[i]=sc.nextLine();
        }
        int sum=0;
        int sSum=0;
        int lSum=0;
        int pass=0;
        for(int i=0;i<sArr.length;i++){
            String tArr[]=sArr[i].split(" ");
            for(int j=1;j<tArr.length;j++){
                sum+=Integer.parseInt(tArr[j]);
            }
            sSum=Integer.parseInt(tArr[2])+Integer.parseInt(tArr[3]);
            lSum=Integer.parseInt(tArr[4])+Integer.parseInt(tArr[5]);
            if(tArr[0].equals("s")){
                if(sum>=350&&sSum>=160){
                    pass+=1;
                }
            }
            if(tArr[0].equals("l")){
                if(sum>=350&&lSum>=160){
                    pass+=1;
                }
            }
            sum=0;
        }
        System.out.print(pass);
        sc.close();
    }
}
