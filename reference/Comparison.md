## コンソール出力
	~~~c#
	Console.WriteLine($"この{hoge}は{hoga}です");
	~~~

	~~~java
	System.out.printf("この%sは%dです",hoge,hoga);
	~~~
	
	~~~js
	console.log('この'+hoge+'は'+hoga+'です');
	~~~
	~~~jsp
	<% System.out.printf("この%sは%dです",hoge,hoga) %>;
	~~~
	~~~python
	print(f'この{hoge}は{hoga}です')
	~~~
