1. 以下のようにパーティーテーブルを作成し、データを挿入する。
	~~~mysql
	create table パーティー(
	id char(3),
	名称 varchar(20),
	職業コード char(2),
	hp int,
	mp int,
	状態コード char(2)
	);

	insert into パーティー values
	('C01','ミナト','01',89,35,'00'),
	('C02','アサカ','11',74,66,'00');
	~~~
